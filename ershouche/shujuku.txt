

user
    -id   int  主键
    -name   varchar(30)
    -idcard  int  身份证
    -logindate  int  登录天数
    -birthday  datetime  生日
    -wechat  varchar(30)  微信
    -qq     int      QQ
    -blance int   账户余额
    -address varchar(100)
    -phonenum  int   手机号
    -loginname   varchar(30)
    -password   varchar(40)
    -sex        单选按钮 -保密(0) -男(1) -女(2)
    -allowbuy     bool 可以买车
    -agreeservice  bool  是否同意服务协议

    -buyserver varchar(100) 购买服务



usercard       银行卡

usedcar

    -carrun     float   车行公里数
    -carengine  varchar(30) 引擎号
    -carfixed   bool   是否维修
    -carimgfront     select
    -carimgside      select
    -carimgbehind    select
    -carprice   float  车价
    -caraddress varchar(100)  卖车地址
    -carprocedures bool  手续齐全

    -carinfo  text  车况

sysmsg
